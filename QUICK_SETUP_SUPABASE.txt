📋 STEP 1: GET SUPABASE CREDENTIALS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Go to: https://supabase.com
2. Sign in or create account
3. Create a new project (or use existing)
4. Go to: Settings → API
5. Copy these two values:
   • Project URL
   • anon public key


📝 STEP 2: CONFIGURE THE APP
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Open file: js/supabase-config.js
2. Replace these lines:

   const SUPABASE_URL = 'YOUR_SUPABASE_URL';
   const SUPABASE_ANON_KEY = 'YOUR_SUPABASE_ANON_KEY';

   With your actual values:

   const SUPABASE_URL = 'https://xxxxx.supabase.co';
   const SUPABASE_ANON_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...';

3. Save the file


🗄️  STEP 3: SET UP DATABASE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. In Supabase Dashboard, go to: SQL Editor
2. Open file: supabase/schema.sql
3. Copy ALL the SQL code
4. Paste into Supabase SQL Editor
5. Click "Run" to create tables


📧 STEP 4: ENABLE EMAIL AUTH
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. In Supabase Dashboard, go to: Authentication → Providers
2. Find "Email" provider
3. Make sure it's ENABLED (toggle should be ON)
4. Save changes


✅ STEP 5: TEST THE APP
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Clear browser cache: Ctrl + Shift + Delete
2. Open: signup.html
3. Fill out the form with a REAL email address
4. Click "Create Account"
5. Check your email for verification link
6. Click the verification link
7. Go to: login.html
8. Log in with your credentials
9. You should be redirected to dashboard!
